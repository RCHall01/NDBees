vcftools --vcf populations.snps.vcf --remove-indels --min-alleles 2 --max-alleles 2 --thin 2 --recode --out BOGR2
vcftools --vcf BOGR2.vcf --max-missing 0.8 --recode --out BOGR2T
plink --vcf BOGR2T.vcf --make-bed --allow-extra-chr  --out BOGR2T
plink --bfile BOGR2T --make-king-table --allow-extra-chr  --out king_relatedness
plink --bfile BOGR2T --missing --allow-extra-chr --out missingness_report
vcftools --vcf BOGR2T.vcf --remove removed_inds.txt --recode --out sib_filt
vcftools --vcf sib_filt.vcf --hwe 0.05 --recode --out BOGR_HWE
plink --vcf BOGR_HWE.vcf --set-all-var-ids @:# --indep-pairwise 50 1 0.2 --allow-extra-chr --out ld_filt
awk -F ":" '{print $1 "\t" $2}' ld_filt.prune.out  > positions.txt
vcftools --vcf BOGR_HWE.vcf --exclude-positions positions.txt --recode --out BOGR_LD
vcftools --vcf BOGR_LD.vcf --depth --out BOGR_depth
