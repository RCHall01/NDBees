vcftools --vcf populations.snps.vcf --remove-indels --min-alleles 2 --max-alleles 2 --thin 2 --recode --out BOTE2
vcftools --vcf BOTE2.vcf --max-missing 0.8 --recode --out BOTE2T
plink --vcf BOTE2T.vcf --make-bed --allow-extra-chr  --out BOTE2T
plink --bfile BOTE2T --make-king-table --allow-extra-chr  --out king_relatedness
plink --bfile BOTE2T --missing --allow-extra-chr --out missingness_report
vcftools --vcf BOTE2T.vcf --remove removed_inds.txt --recode --out sib_filt
vcftools --vcf sib_filt.vcf --hwe 0.05 --recode --out BOTE_HWE
plink --vcf BOTE_HWE.vcf --set-all-var-ids @:# --indep-pairwise 50 1 0.2 --allow-extra-chr --out ld_filt
awk -F ":" '{print $1 "\t" $2}' ld_filt.prune.out  > positions.txt
vcftools --vcf BOTE_HWE.vcf --exclude-positions positions.txt --recode --out BOTE_LD
vcftools --vcf BOTE_LD.vcf --depth --out BOTE_depth
